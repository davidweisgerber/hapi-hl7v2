27  package ca.uhn.hl7v2.util;
28  
29  import ca.uhn.hl7v2.model.Structure;
30  import ca.uhn.hl7v2.util.FilterIterator.Predicate;
31  
32  public class StructureNamePredicate implements Predicate<Structure> {
33  
34  	private final String desired;
35  	
36  	public StructureNamePredicate(String desired) {
37  		super();
38  		this.desired = desired;
39  	}
40  
41  	public boolean evaluate(Structure obj) {
42  		return desired.equals(obj.getName());
43  	}
44  	
45  }
