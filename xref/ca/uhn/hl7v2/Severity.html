27  package ca.uhn.hl7v2;
28  
29  /**
30   * Severity code table
31   */
32  public enum Severity {
33  
34      INFO("I", "Info"),
35      WARNING("W", "Warning"),
36      ERROR("E", "Error");
37  
38      private static final String HL70516 = "HL70516";
39      private final String code;
40      private final String message;
41  
42      Severity(String code, String message) {
43          this.code = code;
44          this.message = message;
45      }
46  
47      /**
48       * @return the string severity code
49       */
50      public String getCode() {
51          return code;
52      }
53  
54      /**
55       * @return the severity code message
56       */
57      public String getMessage() {
58          return message;
59      }
60  
61      /**
62       * Returns the Severity for the given string
63       * @param code severity error code
64       * @return Severity
65       */
66      public static Severity severityFor(String code) {
67          for (Severity severity : Severity.values()) {
68              if (severity.code.equals(code)) {
69                  return severity;
70              }
71          }
72          return null;
73      }
74  
75      /**
76       * @return the HL7 table number
77       */
78      public static String codeTable() {
79          return HL70516;
80      }
81  }
