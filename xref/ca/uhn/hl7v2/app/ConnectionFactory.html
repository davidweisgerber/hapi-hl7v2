26  package ca.uhn.hl7v2.app;

28  import java.io.IOException;
29  import java.net.InetSocketAddress;
30  import java.net.Socket;
31  import java.util.concurrent.ExecutorService;
32  
33  /**
34   * Static Connection factory that creates client-side {@link Connection}s from
35   * {@link ConnectionData} objects.
36   */
37  class ConnectionFactory {
38  
39  	public static Connection open(ConnectionData connectionData,
40  			ExecutorService executorService) throws Exception {
41  
42          return connectionData.isLazy() ?
43                  new LazyConnection(connectionData, executorService) :
44                  openEagerly(connectionData, executorService);
45  
46  	}
47  
48      public static Connection openEagerly(ConnectionData connectionData,
49                                           ExecutorService executorService) throws Exception {
50          Connection connection;
51          if (connectionData.getPort2() == 0) {
52              connection = new ActiveConnection(connectionData.getParser(),
53                      connectionData.getProtocol(), createSocket(connectionData.getSocketFactory(),
54                      connectionData.getHost(), connectionData.getPort(),
55                      connectionData.isTls()), executorService);
56          } else {
57              Socket outbound = createSocket(connectionData.getSocketFactory(), connectionData.getHost(),
58                      connectionData.getPort(), connectionData.isTls());
59              Socket inbound = createSocket(connectionData.getSocketFactory(), connectionData.getHost(),
60                      connectionData.getPort2(), connectionData.isTls());
61              connection = new ActiveConnection(connectionData.getParser(),
62                      connectionData.getProtocol(), inbound, outbound,
63                      executorService);
64          }
65          connection.activate();
66          return connection;
67      }
68  
69  
70  	private static Socket createSocket(ca.uhn.hl7v2.util.SocketFactory socketFactory, String host, int port, boolean tls) throws IOException {
71  		Socket socket;
72  		if (tls) {
73  			socket = socketFactory.createTlsSocket();
74  		} else {
75  			socket = socketFactory.createSocket();
76  		}
77  		socket.connect(new InetSocketAddress(host, port));
78  		return socket;
79  	}
