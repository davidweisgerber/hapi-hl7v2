26  package ca.uhn.hl7v2.validation.impl;
27  
28  import java.util.StringTokenizer;
29  
30  import ca.uhn.hl7v2.validation.MessageRule;
31  
32  /**
33   * A <code>RuleBinding</code> that applies to <code>MessageRule</code>s. 
34   * <p>
35   * In general there is no need to use this class directly. Instead use the
36   * Validation builder classes to generate rules and rule bindings.
37   *  
38   * @author Bryan Tripp
39   */
40  @SuppressWarnings("serial")
41  public class MessageRuleBinding extends RuleBinding<MessageRule> {
42  
43      private final String myMessageType;
44      private final String myTriggerEvent;
45      private static final String delim = "^"; 
46      
47      /**
48       * @param theVersion see {@link RuleBinding#getVersion()}
49       * @param theMessageType a value valid for MSH-9-1 or * for all
50       * @param theTriggerEvent a value valid for MSH-9-2 or * for all
51       * @param theRule rule to be used for binding
52       */
53      public MessageRuleBinding(String theVersion, String theMessageType, String theTriggerEvent, MessageRule theRule) {
54          super(theVersion, theMessageType + "^" + theTriggerEvent, theRule);
55          myMessageType = theMessageType;
56          myTriggerEvent = theTriggerEvent;
57      }
58  
59      /**
60       * @param theScope must be in the form type^event (ie message type and trigger event separated
61       *      by a caret)
62       */
63      public boolean appliesToScope(String theScope) {
64          
65          if (!theScope.matches(".*\\^.*")) {
66              throw new IllegalArgumentException("arg theScope must be of the form type^event");
67          }
68          
69          StringTokenizer tok = new StringTokenizer(theScope, delim, false);
70          String itemType = tok.nextToken();
71          String itemEvent = tok.nextToken();
72          
73          return applies(myMessageType, itemType) && applies(myTriggerEvent, itemEvent);
74      }
75  
76  }
