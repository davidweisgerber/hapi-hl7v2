27  package ca.uhn.hl7v2.llp;
28  
29  import java.io.IOException;
30  import java.io.InputStream;
31  import java.nio.charset.Charset;
32  
33  public class MinLLPReader extends Hl7DecoderReader<MllpDecoder> {
34  
35  	/**
36  	 * @deprecated Use {@link MllpConstants#CHARSET_KEY} instead
37  	 */
38  	@Deprecated()
39      public static final String CHARSET_KEY = MllpConstants.CHARSET_KEY;
40  
41      public MinLLPReader() throws IOException {
42      }
43  
44      public MinLLPReader(InputStream in) throws IOException {
45          super(in);
46      }
47  
48      public MinLLPReader(InputStream in, Charset charset) throws IOException {
49          super(in, charset);
50      }
51  
52      @Override
53      protected MllpDecoder initDecoder() {
54          return new MllpDecoder(getCharset());
55      }
56  
57  
58  }
